## 2.2 set或者map
set和map底层是平衡二叉树或者红黑树，插入之后再遍历其实就是排序的
``` c++
#include <iostream>
#include <string>
#include <stdio.h>
#include <set>
using namespace std;

int main(void) {
    set<int> my_set;
    my_set.insert(11);
    my_set.insert(3);
    my_set.insert(9);
    my_set.insert(1);
    my_set.insert(25);
    for(set<int>::iterator itor = my_set.begin(); itor != my_set.end(); ++itor) {
        cout << iotr << endl;
    }
    return 0;
}
```
